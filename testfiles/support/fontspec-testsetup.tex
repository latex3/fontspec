\newcommand\FIRSTSHIPOUT[2]{%
  % get rid of any gunk due to output hooks etc
  \SETBOX{#1}{#2}%
  \shipout\box#1\relax
  \renewcommand\FIRSTSHIPOUT[2]{}
}

  \OMIT
}
